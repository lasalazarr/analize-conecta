CREATE PROCEDURE [dbo].[PRC_SVDN_PUNTOS_SAVED]
(
	@i_operation						char(1),
	@i_option							char(1),		
	@i_numeropedido						varchar(12)				= NULL,
	@i_nit								varchar(20)				= NULL,
	@i_campana							varchar(4)				= NULL,
	@i_tipo								varchar(2)				= NULL,
	@i_cantidad							int						= NULL,
	@i_movimiento						varchar(50)				= NULL,
	@i_procesado						int						= NULL,
	@o_err_cod							int						OUTPUT,
    @o_err_msg							varchar					OUTPUT
)
AS

BEGIN


	/*================================================================*/
	/*OPERACIONES DE SELECCION                                        */
	/*================================================================*/
	IF @i_operation = 'S'
	BEGIN
			/*================================================================*/
			--Obtiene los puntos efectivos de una empresaria
			IF @i_option = 'A'
			BEGIN

				 --SELECT (puntoEfec-PuntoRed) as PuntoEfec FROM   SVDN_PUNTOS_TOTAL WHERE nit=@i_nit
				  SELECT ISNULL((SELECT (puntoEfec-PuntoRed)  FROM   SVDN_PUNTOS_TOTAL WHERE nit=@i_nit),0) as PuntoEfec
				
			END
	END

	/*================================================================*/
	/*OPERACIONES DE INSERCION                                        */
	/*================================================================*/ 
	IF @i_operation = 'I'
	BEGIN    
		IF @i_option = 'A'
		BEGIN				
		
				
				INSERT INTO SVDN_PUNTOS_DETALLADO
				(       
					Nit						,
					Campana					,
					Documento				,
					Tipo					,
					Cantidad				,
					Fecha					,					
					Movimiento				,
					Procesado		
				) 
				VALUES
				( 	
					@i_nit					,
					(select campana from campanas where fechaini < GETDATE() and GETDATE() < fechafin),
					@i_numeropedido			,
					@i_tipo					,
					@i_cantidad				,
					getdate()				,
					@i_movimiento			,
					@i_procesado			
				)


				--********************************************************************************************************************
				-- SIEMPRE ACTUALIZAR PUNTOS TOTAL

				---------------- Si no existe se crea en puntos total --------------------------------------	
				IF(SELECT COUNT(*) FROM svdn_Puntos_total WHERE nit=@i_nit)=0
				BEGIN

				INSERT INTO svdn_Puntos_total(nit, puntoped, puntoEfec, puntoRed, total)
				VALUES (@i_nit, 0, 0, 0, 0 );
				END
				-------------------END---------------- --------------------------------------	


				UPDATE SVDN_PUNTOS_TOTAL

				SET puntoPed=ISNULL((
				SELECT SUM(cantidad) FROM SVDN_PUNTOS_DETALLADO 
				WHERE nit=@i_nit
				AND procesado=0
				AND tipo='+'),0),

				puntoEfec=ISNULL((
				SELECT SUM(cantidad) FROM SVDN_PUNTOS_DETALLADO 
				WHERE nit=@i_nit
				AND procesado=1
				AND tipo='+'
				),0), 

				puntoRed=ISNULL((
				SELECT SUM(cantidad) FROM SVDN_PUNTOS_DETALLADO 
				WHERE nit=@i_nit			
				AND tipo='-'),0)
				WHERE nit=@i_nit			


				UPDATE SVDN_PUNTOS_TOTAL
				SET total=PuntoPed+puntoEfec-puntoRed
				WHERE nit=@i_nit

			--********************************************************************************************************************
		
		END
		
		IF @i_option = 'B'
		BEGIN		
				
			---------------- Si no existe se crea en puntos total --------------------------------------	
				IF(SELECT COUNT(*) FROM svdn_Puntos_total WHERE nit=@i_nit)=0
				BEGIN

				INSERT INTO svdn_Puntos_total(nit, puntoped, puntoEfec, puntoRed, total)
				VALUES (@i_nit, 0, 0, 0, 0 );
				END
				-------------------END---------------- --------------------------------------	


				UPDATE SVDN_PUNTOS_TOTAL

				SET puntoPed=ISNULL((
				SELECT SUM(cantidad) FROM SVDN_PUNTOS_DETALLADO 
				WHERE nit=@i_nit
				AND procesado=0
				AND tipo='+'),0),

				puntoEfec=ISNULL((
				SELECT SUM(cantidad) FROM SVDN_PUNTOS_DETALLADO 
				WHERE nit=@i_nit
				AND procesado=1
				AND tipo='+'
				),0), 

				puntoRed=ISNULL((
				SELECT SUM(cantidad) FROM SVDN_PUNTOS_DETALLADO 
				WHERE nit=@i_nit			
				AND tipo='-'),0)
				WHERE nit=@i_nit			


				UPDATE SVDN_PUNTOS_TOTAL
				SET total=PuntoPed+puntoEfec-puntoRed
				WHERE nit=@i_nit
			
		END


		--Inserta los puntos adicionales por otros conceptos
  
		IF @i_option = 'C'
		BEGIN				

			DECLARE @VALOR  NUMERIC(18,2)=0
			DECLARE @IVA  NUMERIC(18,2)=0
			DECLARE @TOTAL  NUMERIC(18,2)=0
			DECLARE @VALORPEDIDOMINIMOEXTRA  NUMERIC(18,2)=0
			DECLARE @VALORGANADOPEDIDOMINIMOEXTRA  INT =0
			DECLARE @VALORGANADOPUNTOSSAVED  INT =0

			-----------------------------------------------------------------------------------------------------------------------------
			-- CONCEPTO 10 = POR NIVEL DE PEDIDO

			--PUNTOS POR PEDIDO EXTRA PUNTOS
			SET @VALORPEDIDOMINIMOEXTRA = CAST((SELECT monto FROM SVDN_PUNTOS_NIVELES WHERE id=1) AS NUMERIC)

			SET @VALORGANADOPEDIDOMINIMOEXTRA = CAST((SELECT PUNTOS FROM SVDN_PUNTOS_NIVELES WHERE id=1) AS INT)

			--PRECIO FACTURADO = PRECIO EMPRESARIA
			SET @VALOR= (select sum(valor) from svdn_pedidosc2_2000 where numero=@i_numeropedido and referencia<>'FT0001' )
			SET @IVA = (select sum(valor*(tarifaiva/100)) from svdn_pedidosc2_2000 where numero=@i_numeropedido and referencia<>'FT0001' ) 

			SET @TOTAL= @VALOR + @IVA
				
			-- SI CUMPLE PUNTOS POR NIVEL DE PEDIDO
			IF(@TOTAL >= @VALORPEDIDOMINIMOEXTRA)
			BEGIN
				
				INSERT INTO SVDN_PUNTOS_DETALLADO
				(       
					Nit						,
					Campana					,
					Documento				,
					Tipo					,
					Cantidad				,
					Fecha					,					
					Movimiento				,
					Procesado		
				) 
				VALUES
				( 	
					@i_nit					,
					(select campana from campanas where fechaini < GETDATE() and GETDATE() < fechafin),
					@i_numeropedido								,
					'+'											,
					@VALORGANADOPEDIDOMINIMOEXTRA				,
					getdate()									,
					10											, -- POR NIVEL DE PEDIDO
					0			
				)
			END

			-----------------------------------------------------------------------------------------------------------------------------
			-- CONCEPTO 18 = POR CREACION DE PEDIDO POR SAVED

			SET @VALORGANADOPUNTOSSAVED = CAST((SELECT par_valor FROM SVDN_PARAMETROS WHERE par_id=78) AS INT)

			INSERT INTO SVDN_PUNTOS_DETALLADO
			(       
				Nit						,
				Campana					,
				Documento				,
				Tipo					,
				Cantidad				,
				Fecha					,					
				Movimiento				,
				Procesado		
			) 
			VALUES
			( 	
				@i_nit					,
				(select campana from campanas where fechaini < GETDATE() and GETDATE() < fechafin),
				@i_numeropedido								,
				'+'											,
				@VALORGANADOPUNTOSSAVED						,
				getdate()									,
				18											, -- POR CREACION DE PEDIDO POR SAVED
				0			
			)
			-----------------------------------------------------------------------------------------------------------------------------



			--********************************************************************************************************************
			-- SIEMPRE ACTUALIZAR PUNTOS TOTAL

			---------------- Si no existe se crea en puntos total --------------------------------------	
			IF(SELECT COUNT(*) FROM svdn_Puntos_total WHERE nit=@i_nit)=0
			BEGIN

			INSERT INTO svdn_Puntos_total(nit, puntoped, puntoEfec, puntoRed, total)
			VALUES (@i_nit, 0, 0, 0, 0 );
			END
			-------------------END---------------- --------------------------------------	


			UPDATE SVDN_PUNTOS_TOTAL

			SET puntoPed=ISNULL((
			SELECT SUM(cantidad) FROM SVDN_PUNTOS_DETALLADO 
			WHERE nit=@i_nit
			AND procesado=0
			AND tipo='+'),0),

			puntoEfec=ISNULL((
			SELECT SUM(cantidad) FROM SVDN_PUNTOS_DETALLADO 
			WHERE nit=@i_nit
			AND procesado=1
			AND tipo='+'
			),0), 

			puntoRed=ISNULL((
			SELECT SUM(cantidad) FROM SVDN_PUNTOS_DETALLADO 
			WHERE nit=@i_nit			
			AND tipo='-'),0)
			WHERE nit=@i_nit			


			UPDATE SVDN_PUNTOS_TOTAL
			SET total=PuntoPed+puntoEfec-puntoRed
			WHERE nit=@i_nit

		--********************************************************************************************************************
		END
	END
	
END
